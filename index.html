  <h1>Roof Estimate Calculator</h1>

  <form id="estimateForm">

    <label for="stories">How many stories?</label>

    <input type="number" id="stories" name="stories" min="1" required />


    <label for="steepness">Steepness of the roof:</label>

    <select id="steepness" name="steepness" required>

      <option value="">-- Select Steepness --</option>

      <option value="low">Low</option>

      <option value="medium">Medium</option>

      <option value="high">High</option>

    </select>


    <label for="squareFootage">Total square footage:</label>

    <input type="number" id="squareFootage" name="squareFootage" min="100" required />


    <button type="submit">Calculate Estimate</button>

  </form>


  <div id="estimate"></div>


  <script>

    document.getElementById("estimateForm").addEventListener("submit", function(event) {

      event.preventDefault();


      const stories = parseInt(document.getElementById("stories").value);

      const steepness = document.getElementById("steepness").value;

      const squareFootage = parseInt(document.getElementById("squareFootage").value);


      const estimateDiv = document.getElementById("estimate");


      // Show message for buildings taller than 3 stories

      if (stories > 3) {

        estimateDiv.innerHTML = `<span class="note">Thank you for using our estimator tool. For buildings taller than 3 stories, please contact our office directly to receive a custom estimate tailored to your project.</span>`;

        return;

      }


      let baseRate = 0.06; // $0.06 per square foot

      let estimate = squareFootage * baseRate;


      // Stories markup

      if (stories === 2) {

        estimate *= 1.20; // 20% increase

      } else if (stories === 3) {

        estimate *= 1.30; // 30% increase

      }


      // Steepness markup with 3-story logic

      if (steepness === "low") {

        if (stories === 3) estimate *= 1.10;

      } else if (steepness === "medium") {

        estimate *= (stories === 3) ? 1.20 : 1.10;

      } else if (steepness === "high") {

        estimate *= (stories === 3) ? 1.40 : 1.20;

      }


      estimateDiv.innerHTML = `Estimated Price: $${estimate.toFixed(2)}`;

    });

  </script>
